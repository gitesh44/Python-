# Creating a dictionary

# what is dictionary?
--->  o  A dictionary in Python is a collection of key-value pairs.
      o  Each key is unique and maps to a value.
      o  Dictionaries are mutable and unordered.
o In Python, a Dictionary is an unordered, mutable, and indexed collection of key-value pairs.
o Keys must be unique and immutable (e.g., string, number, tuple), while 
  values can be of any data type (mutable or immutable).
  
  my_dict = {
    "name": "Gitesh",
    "age": 22,
    "skills": ["Python", "Java", "SQL"]
}



my_dict = {
    'name': 'Alice', 
    'age': 25,
     'city': 'New York'
     
     }

# Accessing values
print(my_dict['name'])  
# Output: Alice

# Adding a new key-value pair
my_dict['email'] = 'alice@example.com'

# Updating a value
my_dict['age'] = 26

# Removing a key-value pair
del my_dict['city']

# Iterating over keys and values
for key, value in my_dict.items():
    print(key, value)

# Checking if a key exists
if 'name' in my_dict:
    print('Name exists!')
